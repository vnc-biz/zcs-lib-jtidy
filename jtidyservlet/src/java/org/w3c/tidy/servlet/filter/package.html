Use this filter instead of TidyTag if you don't want to modify your JSP pages or HTML is denerated by non JSP servlets. Fragment from web.xml to enable this Filter. 
<web-app>
    ...
    <filter>
        <filter-name>JTidyFilter</filter-name>
        <filter-class>org.w3c.tidy.servlet.filter.JTidyFilter</filter-class>
        <init-param>
            <param-name>properties.filename</param-name>
            <param-value>MyJTidyServlet.properties</param-value>
        </init-param>
        <init-param>
            <param-name>tee</param-name>
            <param-value>false</param-value>
        </init-param>
        <init-param>
            <param-name>config</param-name>
            <param-value>indent: auto; indent-spaces: 2</param-value>
        </init-param>
        <init-param>
            <param-name>doubleValidation</param-name>
            <param-value>false</param-value>
        </init-param>
        <init-param>
            <param-name>validateOnly</param-name>
            <param-value>false</param-value>
        </init-param>
        <init-param>
            <param-name>doubleValidation</param-name>
            <param-value>false</param-value>
        </init-param>
    </filter>

    <filter-mapping>
        <filter-name>JTidyFilter</filter-name>
        <url-pattern>*.jsp</url-pattern>
    </filter-mapping>
    ...
</web-app>
